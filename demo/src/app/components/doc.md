### Module Usage
```typescript
import { NgModule } from '@angular/core';
import { ngfModule } from "angular-file";

@NgModule({imports: [ ngfModule ]})
export class AppModule {}
```

### Annotations
```typescript
@Directive({ selector: '[ngf]' })
@Directive({ selector: '[ngfSelect]' })
@Directive({ selector: '[ngfDrop]' })
```
